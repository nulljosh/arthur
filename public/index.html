Architecture

Mixture of Experts with RoPE, GQA, and RMSNorm

Transformer Forward Pass

def forward(self, x, mask=None):
    for layer in self.layers:
        # RMSNorm + Multi-Head Attention (GQA)
        h = layer.norm1(x)
        h = layer.attention(h, mask=mask)
        x = x + h
        # RMSNorm + MoE Feed-Forward
        h = layer.norm2(x)
        h = layer.moe(h)  # top-2/4 expert routing
        x = x + h
    return self.head(self.norm_f(x))
